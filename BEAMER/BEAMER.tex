\begin{frame}{\textsc{Objectives:}}
    The paper {\color{primaryred}"Empirical Asset Pricing via Machine Learning"} has a twofold objective \cite{abadie2003economic}
  \begin{enumerate}
   \item Setting a {\color{steelblue} new benchmarks} to compare ML methods for asset pricing.
   \item Synthesizing the {\color{steelblue}asset pricing theory} with the emerging {\color{steelblue} ML literature}.
  \end{enumerate}
\end{frame}

\begin{frame}{\textsc{Definition:}}
    What is {\color{steelblue}\textbf{Machine Learning (ML)}}?
    \begin{enumerate}
        \item {\color{primaryred}High-dimensional models.}
        \begin{itemize}
            \item[\textcolor{darkgreen}{\ding{51}}] Flexibility.
            \item[\textcolor{red}{\ding{55}}] Overfitting.
        \end{itemize}
        \item {\color{primaryred}“Regularization” methods for model selection.}
            \begin{itemize}
                \item[\textcolor{darkgreen}{\ding{51}}] Fix Overfitting.
            \end{itemize}
        \item {\color{primaryred}Efficient algorithms for searching model specifications.}
            \begin{itemize}
                \item[\textcolor{darkgreen}{\ding{51}}] Fix non-linearities.
            \end{itemize}
    \end{enumerate}
\end{frame}


\begin{frame}{\textsc{Dataset and Metrics:}}
    \begin{itemize}
        \item {\color{primaryred}\textbf{Universe:}} Monthly U.S. equities (CRSP/Compustat)
        \item {\color{primaryred}\textbf{Sample:}} {\color{steelblue}\textbf{30,000 individual stocks}}.
        \item {\color{primaryred}\textbf{Period:}} {\color{steelblue}\textbf{60 years}} (1957–2016)
        \item {\color{primaryred}\textbf{Predictors:}} {\textbf{94 firm-level characteristics}} from the "factor zoo" + {\textbf{74 industry-sector dummies}} + interaction terms.

        \end{itemize}
    \end{itemize}
