\begin{frame}{\textsc{The Brazilian Paradox, Taxation and Inequality:}}

  \textsc{\color{harmonizedblue}Context:} \\
  \medskip
  Since the colonial period, Brazil has been characterized by high levels of income inequality:
  \begin{itemize}
    \item \textcolor{primaryred}{Gini}: 0.51 (\textcolor{steelblue}{5th highest globally}) (2023).
    \item \textcolor{primaryred}{Wealth concentration:} Top 1\% owns \textcolor{steelblue}{48.4\%} of total wealth (\textcolor{steelblue}{1st globally}) (2022).
    \item \textcolor{primaryred}{Regressive tax system:} \textcolor{steelblue}{42.5\%} average tax rate vs. \textcolor{steelblue}{20.6\%} for million-dollar earners (Palomo et al.\cite{zucmanpalomo}).
  \end{itemize}
  \textsc{\color{harmonizedblue}And in 1995...} \\
  \medskip
  \begin{center}
    \large
    Dividend tax cut: \textcolor{primaryred}{25\%} ${\color{steelblue} \Rightarrow}$ \textbf{\color{primaryred}0\%}
  \end{center}

\end{frame}


\begin{frame}{\textsc{Research Question \& Literature :}}

  \begin{center}
   \textit{Did the 1995 \textcolor{primaryred}{dividend tax cut} in Brazil lead to an increase in \textcolor{primaryred}{top 1\% income inequality}?}
  \end{center}
  \medskip
  \textsc{\color{harmonizedblue}Theoretical Framework:}
  \begin{enumerate}
    \item \textcolor{primaryred}{Saez-Chetty Agency Model} \cite{chetty_agency_2007} and \textcolor{primaryred}{Suarez-Zidar Static Spatial Equilibrium Model} \cite{suarezetzidar}: Dividend tax cuts ${\color{steelblue}\Rightarrow}$  increase payouts \& capital income ${\color{steelblue}\Rightarrow}$  increase in capital-to-labor income inequality.   
  \end{enumerate}  
  \medskip
  \textsc{\color{harmonizedblue}Key Studies:}
    \begin{enumerate}
      \item \textcolor{primaryred}{Nallareddy et al.} \cite{nallareddy2021corporate}: \textcolor{harmonizedblue}{effect of corporate income on income inequality} in the US using \textcolor{steelblue}{DiD + Synthetic Control Method (SCM)}.
      \item \textcolor{primaryred}{Berman and Klor} \cite{berman2024capital}: \textcolor{harmonizedblue}{effect 2017 dividend tax cut on top-income inequality} in Israel.
    \end{enumerate}
\end{frame}

\begin{frame}{\textsc{Database \& Methods:}}

  \textsc{\color{harmonizedblue}Database:}
  \begin{itemize}
      \item \textcolor{primaryred}{WID Database:} Provides the top 1\% income shares for several countries.
      \item \textcolor{primaryred}{World Bank \& PennProject:} Control variables (GDP growth, unemployment, inflation, etc.).
      \item \textcolor{primaryred}{Morgan and Souza:} \cite{morgan2025distribution} Complementary data for Brazilian income shares.
  \end{itemize}

  \medskip
  \textsc{\color{harmonizedblue}Total:} 35 variables + interactions, 30 years, 50+ countries.
  \medskip

  \textsc{\color{harmonizedblue}The SCM Approach:}
  \begin{itemize}
      \item Due to \textcolor{primaryred}{Abadie and Gardeazabal} \cite{abadie2003economic} and further developed in \textcolor{primaryred}{Abadie et al.} \cite{abadie2010synthetic}.
      \item \textcolor{harmonizedblue}{... with ML:}
      \begin{itemize}
        \item[\textcolor{darkgreen}{\ding{51}}] Helps with \textcolor{steelblue}{overfitting and sparsity} (e.g. Lasso regularization- see \textcolor{primaryred}{Abadie and L'Hour} \cite{abadie_penalized_2021}).
        \item[\textcolor{darkgreen}{\ding{51}}] Accounts for potential \textcolor{steelblue}{non-linearities} in predictor-outcome relationships ( Araujo \cite{araujo2023synthetic}).
        \item[\textcolor{darkgreen}{\ding{51}}] Offers a nice way to test \textcolor{steelblue}{robustness} of results.
      \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{A Synthetic Control Method:}
  \begin{align*}
    {\color{primaryred}\min_{w}} \quad 
    {\color{primaryred}\| y_{1,\text{pre}} - X_{0,\text{pre}} w \|^2}
    \\
    \text{s.t.} \quad 
    {\color{steelblue}w_j \ge 0 \quad \forall j,}
    \quad 
    {\color{harmonizedblue}\sum_{j=1}^{J} w_j = 1}
  \end{align*}
\end{frame}

\begin{frame}{The ASCM}
  \begin{align*}
    {\color{primaryred}w^{\text{ASCM}}}
    =
    {\color{steelblue}w^{\text{SCM}}}
    +
    {\color{primaryred}
    Z_0'
    \left(
    Z_0 Z_0'
    + {\color{harmonizedblue}\lambda I}
    \right)^{-1}
    \left(
    Z_1 - Z_0 w^{\text{SCM}}
    \right)}
    \end{align*}
\end{frame}

\begin{frame}{\textsc{Other Specifications:}}
  Potentially, I would like to try other specifications
  \begin{itemize}
    \item \textcolor{primaryred}{Clustering + Random Forest}: From Araujo \cite{araujo2023synthetic}, allows to capture non-linearities and interactions without having to specify them a priori.
    \item \textcolor{primaryred}{Neural Networks:} Very flexible, can capture complex relationships, but less interpretable and may require more data.
    \item \textcolor{primaryred}{Matrix Completion (alternative method)}: Uses a global low-rank structure and learns latent factors across all units. Possible alternative method but less anchored in the dependent variable.
  \end{itemize}
  
\end{frame}

  \begin{frame}{Interpreting the Descriptive Statistics}
  
  \textbf{Main findings}
  
  \begin{itemize}
      \item Digital multinationals are highly heterogeneous in terms of size and organizational structure
      \item On average, groups operate in more than \textbf{34 countries}, confirming their strong international reach
      \item Ownership structures can be very deep, with a maximum of \textbf{10 layers}
  \end{itemize}
  
  \textbf{Tax haven exposure}
  
  \begin{itemize}
      \item Tax haven use is widespread: nearly \textbf{88\%} of groups have at least one haven affiliate
      \item However, the \textbf{intensity} of tax haven use varies strongly across firms
  \end{itemize}
  
  \textbf{Implication}
  
  \begin{itemize}
      \item Complexity is not one-dimensional: it combines size, hierarchy, geography, and possibly fiscal organization
  \end{itemize}
  
  \end{frame}
  
  %------------------------------------------------
  \begin{frame}{Methodology --- Purpose of the Complexity Index}
  
  \textbf{Research objective}
  
  \begin{itemize}
      \item Construct an index capturing the \textbf{structural complexity} of digital multinational groups
      \item Use this index as a proxy for firms' \textbf{capacity to engage in tax avoidance}
  \end{itemize}
  
  \medskip
  
  \textbf{Important distinction}
  
  \begin{itemize}
      \item The index does \textbf{not} directly measure tax avoidance
      \item It captures structural characteristics that may \textbf{facilitate} tax optimization strategies
  \end{itemize}
  
  \medskip
  
  \textbf{Empirical strategy}
  
  \begin{itemize}
      \item Build the index at the group level using the full network
      \item Test its relationship with tax-related outcomes in the financial sample
  \end{itemize}
  
  \end{frame}
  
  %------------------------------------------------
  \begin{frame}{Methodology --- PCA Approach}
  
  \textbf{Why PCA?}
  
  \begin{itemize}
      \item All the variables I selected are variables that shows the structural complexity of a firm
      \item PCA allows me to extract this \textbf{common component} across these dimensions
      \item It avoids imposing arbitrary equal weights
  \end{itemize}
  
  
  \textbf{Variables included in the PCA}
  
  \begin{itemize}
      \item Log number of subsidiaries
      \item Maximum ownership depth
      \item Log number of countries
      \item Geographic dispersion $(1-\text{HHI})$
      \item Sectoral diversification
  \end{itemize}
  
  \textbf{Interpretation}
  
  \begin{itemize}
      \item The first principal component is interpreted as a measure of \textbf{structural complexity}
  \end{itemize}
  
  \end{frame}
  
  %------------------------------------------------
